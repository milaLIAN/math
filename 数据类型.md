~~~python
# 数据类型
# 整数类型：int
# 浮点数类型：float
# 布尔类型：bool(True、False)
# 字符串类型：string

# 整数类型
# 整数可以表示为二进制、十进制、八进制和十六进制
# 默认是十进制
# 二进制：0b
# 八进制：0o
# 十六进制：0x

print(0b1010)
print(0x10)
print(0o12)

# 输出结果
10
16
10

# 浮点数
# 浮点数由整数部分和小数部分组成
# 浮点数存储不精确性
# 因为数据在计算机中存储的方式是二进制的形式
# 所以浮点数计算的时候会出现精确性的问题

print(1.1+2.2)

# 输出结果
# 3.3000000000000003
# 并不是所有的浮点数计算都会出现上述的问题，只有少数
#解决方法
from decimal import Decimal
print(Decimal('1.1')+Decimal('2.2'))

# 结果
# 3.3 结果正常

# 布尔类型
f1=True
f2=False
print(f1+1)
print(f2+1)

# 输出结果
# 2
# 1
# 布尔类型可以直接进行加减运算

# 字符串类型
# 使用单引号或双引号只能在一行使用
# 使用三个单引号或三个双引号可以在多行进行使用
str1='人生苦短'
str2="人生苦短"
str3="""人生苦短"""
str4='''人生苦短'''
print(str1,type(str1))
print(str2,type(str2))
print(str3,type(str3))
print(str4,type(str4))

# 结果
人生苦短 <class 'str'>
人生苦短 <class 'str'>
人生苦短 <class 'str'>
人生苦短 <class 'str'>

# 类型转换
# 讲不听的数据类型拼接在一起
# 可以进行数据类型转换的函数

# str() 将其他数据类型转化成字符串 
#       可以用引号转换 str(123) str("123")

# int() 文字类和小数类字符串，无法转化成整数 int('123')
#       浮点数转化成整数，抹零取整 int(9.8)
#       字符串试图转换成整数时，字符串必须是数字串，还必须是整数串
#       其他类型可转换成

# float() 文字类无法转成整数 float('9.9')
#         整数转成浮点数，末尾为0 float(9)
#         其他类型可转换
~~~
