~~~python
# while循环
while 条件表达式：
#      条件执行体

# 当条件表达式为True的时候，就会一直进行while操作
# 练习1-100偶数和
sum=0
a=0
while a<=100:
    if(a%2==0):
        sum+=a
    a+=1
print(sum)

# for-in循环
# in:从（字符串、序列等）中依次取值
# for-in遍历的对象必须是可迭代对象

for 自定义的变量 in 可迭代对象：
    循环体


# 循环体内不需要访问自定义变量，可以将自定义变量替代为下划线
# 将可迭代对象中的值一个一个取出来，然后依次赋值给自定义变量
for item in range(10):
    # 依次取出并赋值
    print(item)
# 循环中不需要使用自定义变量，将自定义变量写为"_"
# 依次打印python
for _ in range(10):
    print("python")
# 使用for-in来计算0-100内的所有偶数和
sum=0
for item in range(0,100):
    if item%2==0:
        sum+=item
    item+=1
print("sum=",sum)

# 输出100-999之间的所有水仙花数
for item in range(100,1000):
    ge=item%10
    shi=item//10%10
    bai=item//100
    if ge**3+shi**3+bai**3==item:
        print(item)

# break:退出当前分支,与if分支进行使用
# continue:用于结束当前循环，进入下一次循环，通常与分支结构中的if一起使用
~~~
