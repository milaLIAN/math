##### 特殊属性

```python
class A:
    pass
class B:
    pass
class C(A,B):
    def __init__(self,name,age):
        self.name = name
        self.age = age

    def show(self   ):
        print(self.name, self.age)

x=C('Marry',16)
# 查看实例对象的属性字典
print(x.__dict__)
# 查看类的属性字典
print(C.__dict__)
# 查看实例对象所属的类
print(x.__class__)
# 查看父类类型元组
#print(x.__bases__)
# 查看第一个继承的父类
#print(x.__base__)
# 查看类的层次结构
print(C.__mro__)
# 查看子类的列表
print(A.__subclasses__())
# {'name': 'Marry', 'age': 16}
# {'__module__': '__main__', '__init__': <function C.__init__ at 0x00000234BF0FEEA0>, 'show': <function C.show at 0x00000234BF0FEF28>, '__doc__': None}
# <class '__main__.C'>
# (<class '__main__.C'>, <class '__main__.A'>, <class '__main__.B'>, <class 'object'>)
# [<class '__main__.C'>]
```

##### 特殊方法

- 针对类的运算

```python
class Student:
    def __init__(self,name):
        self.name = name
    def __add__(self,other):
        # 通过自定义的方式使得不同的类属性可以相加
        return self.name + other.name
    def __len__(self):
        # self类型的数据无需传参
        return len(self.name)
    # 通过内置函数len()的方式使得可以获得类的属性的长度
stu1=Student('zhangsan')
stu2=Student('wangwei')
print(stu1.__add__(stu2))
print(stu1.__len__())

# __new__()用于创建对象
# __init__()用于对用例进行初始化
```
