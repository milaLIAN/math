~~~python
# \ +转义功能的首字母
# \n 换行
# \t 制表符,中间是4个单元格
print("hello\nworld")
print("hello\tworld")

# \r 回车，对前面的内容进行了覆盖
# \b 退一个格
# \\ 可以转出来一个\
# 单引号中包含单引号
print('老师说：\'大家好\'')

# 原字符，不希望字符串中的转义字符起作用，就使用原字符，就是在字符串之前加上r或R
# 就是相当于将把转义字符的含义给隐身掉
print(r'hello\nworld')
# 注意事项，最后一个字符不能是反斜杠
~~~
