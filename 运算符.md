~~~python
# 算术运算符
# 加减乘除
# 取整
# 取余
# 幂
a=11
b=2
print(a+b)
print(a-b)
print(a*b)
print(a/b)

# 取整的时候需要向下取整
print(a//b)
print(a**b)

print(a % b)

# 结果
# 13
# 9
# 22
# 5.5
# 5
# 121
# 1

# 赋值运算符=
# 执行顺序：从右向左
# 打印结果10
b=9
a=b+1
print(a)

# 支持链式赋值
# 打印结果10 10 10
a=b=c=10
print(a,b,c)

# 支持参数赋值
# +=、-=、*=、/=
# 当发生除法的时候，有可能出现整数转小数的情况

# 支持系列解包赋值
# 要求赋值左右两边的数量相同，否则会发生错误
# 赋值左右两边一一对应
a,b,c=10,20,30

# 交换表达式
a,b=b,a


# 比较运算符
# 比较运算符的结果是布尔类型True、False
# 关于>=、<=的结果只要有一个结果是正确的，就是True
# 比较值的运算符==
# 比较表示is，即两者的id是否相同，结果依然是布尔类型
# ==可以比较数组
arr1=[]
arr2=[]
print(arr1==arr2)
# 不等运算符is not

# 布尔运算符
and=="和"
or=="或者"
not=="非"
in=="在不在当中"
not in=="不在其中"

# 位运算符
# & 按位与：二进制的位同为1时才为1
# | 按位或：二进制的位同为0时才为0
# << 左移  4<<2 向左移动两位，相当于×
# >> 右移  4>>2 向右移动两位，相当于÷

# 运算符的优先级
# 算术运算符：正常按照常识
# 位运算符
# 比较运算符
# 布尔运算符
# 赋值运算符
~~~
